export function greeting(content, name, location, age) {
    let today = new Date();
    let year = today.getFullYear();
    content += location === 'es' ? '!Hola' : 'Hello';
    content += ' ' + name.toUpperCase() + '! ';
    content += ' This year of ' + year + ' ';
    content += 'you are ' + age + ' years old.';
    return content;
}

//USAGE: greeting('Greeting: ', 'Fulano', 'es', '22');

The test

              
it("won't work, good luck", function() {
  let result = greeting('Greeting: ', 'Fulano', 'es', '27');
  expect(result).toEqual('???');
})

Nailed it!

              
//Before each
const DATE_TO_USE = new Date('2016');
const _Date = Date;
global.Date = jest.fn(() => DATE_TO_USE);
global.Date.UTC = _Date.UTC;
global.Date.parse = _Date.parse;
global.Date.now = _Date.now;
